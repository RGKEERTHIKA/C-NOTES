loop in c++                                                                   
loop in c++ are used to execute a block of code repeatedly unit a certain condition is met .They help in manage repetitive tasks.
 
1.FOR LOOP  : used when the number of iterations is know beforehand.
2.WHILE LOOP : used when the number of iterations is not know and depend on a condition 
3.DO-WHILE LOOP: similar to the while loop , but guarantees that the block of code will be executed at least once before checking the condition 

1. //printing numbers from 1 to 5 using for loop 
#include<iostream>
using namespace std;

int main ()
{
    for( int i =1 ; i<=5 ; i++)
    {
        cout<<i<<endl;
    }
return 0;
}

2. //printing numbers from 1 to 5 using while loop 
#include<iostream>
using namespace std;

int main ()
{
   int i =1;
   
    while (i<=5)
   {
       cout<<i<<endl;
       i++;
   }
return 0;
}     

3. //printing numbers from 1 to 5 using do-while loop 
#include<iostream>
using namespace std;

int main ()
{
   int i =1;
   
    while (i<=5)
   {
       cout<<i<<endl;
       i++;
   }
   while (i<= 5);
return 0;
}
